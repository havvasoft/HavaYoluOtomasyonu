
USE HavayoluOtomasyonDB;
GO


CREATE TABLE Airports (
    AirportID INT IDENTITY(1,1) PRIMARY KEY,
    AirportName NVARCHAR(150) NOT NULL,
    City NVARCHAR(100) NOT NULL,
    Country NVARCHAR(100) NOT NULL,
    IATACode VARCHAR(3) UNIQUE NOT NULL,
    TimeZone VARCHAR(50)
);

CREATE TABLE Models (
    ModelID INT IDENTITY(1,1) PRIMARY KEY,
    Manufacturer NVARCHAR(100) NOT NULL,
    ModelName NVARCHAR(100) NOT NULL,
    ICAO_TypeCode VARCHAR(10) NOT NULL,
    Capacity_Economy INT NOT NULL,
    Capacity_Business INT NOT NULL,
    AvgSpeed DECIMAL(6,2)
);

CREATE TABLE Roles (
    RoleID INT IDENTITY(1,1) PRIMARY KEY,
    RoleName NVARCHAR(100) NOT NULL,
    Department NVARCHAR(100),
    SalaryGrade VARCHAR(10)
);

CREATE TABLE Fares (
    FareID INT IDENTITY(1,1) PRIMARY KEY,
    FareBasicCode VARCHAR(20) NOT NULL,
    Refundable BIT NOT NULL, 
    Changeable BIT NOT NULL,
    CabinType VARCHAR(20) NOT NULL
);

CREATE TABLE Passengers (
    PassengerID INT IDENTITY(1,1) PRIMARY KEY,
    FirstName NVARCHAR(100) NOT NULL,
    LastName NVARCHAR(100) NOT NULL,
    Email VARCHAR(150) UNIQUE,
    PassportNumber VARCHAR(20) UNIQUE NOT NULL,
    Phone VARCHAR(20),
    DateOfBirth DATE NOT NULL,
    Gender VARCHAR(10),
    Nationality NVARCHAR(50),
    LoyaltyProgramID VARCHAR(50),
    EmergencyContact VARCHAR(150),
    SpecialNeeds NVARCHAR(255)
);



CREATE TABLE Aircrafts (
    AircraftID INT IDENTITY(1,1) PRIMARY KEY,
    ModelID INT FOREIGN KEY REFERENCES Models(ModelID),
    ManufactureDate DATE,
    TailNumber VARCHAR(20) UNIQUE NOT NULL,
    Status VARCHAR(20) NOT NULL
);

CREATE TABLE Routes (
    RoutesID INT IDENTITY(1,1) PRIMARY KEY,
    DepartureAirportID INT FOREIGN KEY REFERENCES Airports(AirportID),
    ArrivalAirportID INT FOREIGN KEY REFERENCES Airports(AirportID),
    Distance DECIMAL(10,2),
    EstimatedDuration INT
);

CREATE TABLE Staff (
    StaffID INT IDENTITY(1,1) PRIMARY KEY,
    FirstName NVARCHAR(100) NOT NULL,
    LastName NVARCHAR(100) NOT NULL,
    Email VARCHAR(150) UNIQUE,
    Phone VARCHAR(20),
    HireDate DATE NOT NULL,
    RoleID INT FOREIGN KEY REFERENCES Roles(RoleID),
    EmploymentStatus VARCHAR(20) 
);

CREATE TABLE Gates (
    GateID INT IDENTITY(1,1) PRIMARY KEY,
    AirportID INT FOREIGN KEY REFERENCES Airports(AirportID),
    GateNumber VARCHAR(10) NOT NULL,
    Terminal VARCHAR(50)
);


CREATE TABLE Flights (
    FlightID INT IDENTITY(1,1) PRIMARY KEY,
    RoutesID INT FOREIGN KEY REFERENCES Routes(RoutesID),
    AircraftID INT FOREIGN KEY REFERENCES Aircrafts(AircraftID),
    DepartureTime DATETIME NOT NULL,
    ArrivalTime DATETIME NOT NULL,
    FlightStatus VARCHAR(20)
);

CREATE TABLE TechnicalCertifications (
    CertificationID INT IDENTITY(1,1) PRIMARY KEY,
    StaffID INT FOREIGN KEY REFERENCES Staff(StaffID),
    ModelID INT FOREIGN KEY REFERENCES Models(ModelID),
    LicenseType VARCHAR(50) NOT NULL,
    ExpiryDate DATE NOT NULL
);

CREATE TABLE MaintenanceLogs (
    LogID INT IDENTITY(1,1) PRIMARY KEY,
    AircraftID INT FOREIGN KEY REFERENCES Aircrafts(AircraftID),
    LeadEngineerID INT FOREIGN KEY REFERENCES Staff(StaffID),
    MaintenanceType VARCHAR(50) NOT NULL,
    Description NVARCHAR(MAX),
    MaintenanceDate DATE NOT NULL
);

CREATE TABLE Bookings (
    BookingID INT IDENTITY(1,1) PRIMARY KEY,
    PNR_Code VARCHAR(6) UNIQUE NOT NULL,
    BookingDate DATETIME DEFAULT GETDATE(),
    TotalAmount DECIMAL(18,2) NOT NULL,
    PaymentStatus VARCHAR(20) NOT NULL,
    CreatedByStaff INT FOREIGN KEY REFERENCES Staff(StaffID) NULL
);


CREATE TABLE Tickets (
    TicketID INT IDENTITY(1,1) PRIMARY KEY,
    TicketNumber VARCHAR(20) UNIQUE NOT NULL,
    FlightID INT FOREIGN KEY REFERENCES Flights(FlightID),
    PassengerID INT FOREIGN KEY REFERENCES Passengers(PassengerID),
    SeatNumber VARCHAR(5),
    ClassType VARCHAR(20) NOT NULL,
    BookingID INT FOREIGN KEY REFERENCES Bookings(BookingID),
    FareID INT FOREIGN KEY REFERENCES Fares(FareID),
    IsCheckedIn BIT DEFAULT 0,
    BaggageAllowance INT 
);

CREATE TABLE Payments (
    PaymentID INT IDENTITY(1,1) PRIMARY KEY,
    BookingID INT FOREIGN KEY REFERENCES Bookings(BookingID),
    PaymentMethod VARCHAR(50),
    TransactionID VARCHAR(100) UNIQUE,
    Amount DECIMAL(18,2) NOT NULL,
    PaymentDate DATETIME DEFAULT GETDATE()
);

CREATE TABLE Baggage (
    BaggageID INT IDENTITY(1,1) PRIMARY KEY,
    TicketID INT FOREIGN KEY REFERENCES Tickets(TicketID),
    Weight DECIMAL(5,2) NOT NULL,
    BaggageTagNumber VARCHAR(50) UNIQUE,
    Status VARCHAR(20) 
);

CREATE TABLE FlightStatusHistory (
    HistoryID INT IDENTITY(1,1) PRIMARY KEY,
    FlightID INT FOREIGN KEY REFERENCES Flights(FlightID),
    Status VARCHAR(20) NOT NULL,
    Reason NVARCHAR(255),
    UpdateTime DATETIME DEFAULT GETDATE()
);

CREATE TABLE FlightCrew (
    FlightCrewID INT IDENTITY(1,1) PRIMARY KEY,
    FlightID INT FOREIGN KEY REFERENCES Flights(FlightID),
    StaffID INT FOREIGN KEY REFERENCES Staff(StaffID),
    AssigmentRole VARCHAR(50) NOT NULL 
);
